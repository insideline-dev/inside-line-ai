│ Update PdfProcessor                                                          │
│ (backend/src/modules/analysis/processors/pdf.processor.ts):                  │
│ import { generateMemoTemplate } from                                         │
│ '../../startup/templates/memo.template';                                     │
│ import PDFDocument from 'pdfkit';                                            │
│                                                                              │
│ private async generateAndUploadPdf(                                          │
│   startupData: Startup,                                                      │
│   scores: StartupScores | null,                                              │
│   userId: string,                                                            │
│ ): Promise<{ pdfUrl: string; pdfKey: string }> {                             │
│   // Create PDF document                                                     │
│   const doc = new PDFDocument();                                             │
│   const chunks: Buffer[] = [];                                               │
│                                                                              │
│   doc.on('data', (chunk) => chunks.push(chunk));                             │
│                                                                              │
│   // Use template                                                            │
│   generateMemoTemplate(doc, startupData, scores);                            │
│                                                                              │
│   doc.end();                                                                 │
│                                                                              │
│   // Wait for PDF to finish                                                  │
│   const pdfBuffer = await new Promise<Buffer>((resolve) => {                 │
│     doc.on('end', () => resolve(Buffer.concat(chunks)));                     │
│   });                                                                        │
│                                                                              │
│   // Upload to storage                                                       │
│   const result = await this.storageService.uploadGeneratedContent(           │
│     userId,                                                                  │
│     'documents',                                                             │
│     pdfBuffer,                                                               │
│     'application/pdf',                                                       │
│     startupData.id,                                                          │
│     {                                                                        │
│       startupId: startupData.id,                                             │
│       startupName: startupData.name,                                         │
│       generatedAt: new Date().toISOString(),                                 │
│     },                                                                       │
│   );                                                                         │
│                                                                              │
│   return { pdfUrl: result.url, pdfKey: result.key };                         │
│ }                                                                            │

│ Promise<WebsiteMetadata> {                                                   │
│   try {                                                                      │
